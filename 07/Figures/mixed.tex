\begin{tikzpicture}
    \graph[feynman electrical layout, horizontal=in to out]
    {
        {[edges=fermion]
            in -- a -- b -- c -- out;
        };
    };

    \coordinate (middle1) at ($(a)!.5!(b)$);
    \coordinate (middle2) at ($(b)!.5!(c)$);

    \vertex[draw, shape=rectangle, above=2cm of middle1] (i) {};
    \vertex[draw, shape=rectangle, above=2cm of c] (j) {};

    \draw[scalar, momentum'={$\vec q_1$}] (a) to  (i);
    \draw[scalar, momentum={$\vec q_2$}, momentum/shorten=10mm] (b) to  (j);
    \draw[scalar, momentum'={$\vec q_3$}, momentum/shorten=13mm]  (c) to (i);
\end{tikzpicture}
